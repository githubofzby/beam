# 总批处理命令
```bash
THREADS=40
BEAM=./build/beam
DATA=/data/zby/datasets

TOPK=16
ITERS=20
SEED=1
ERROR_BOUND=0.0
GROUP_BATCH_SIZE=64
CANDIDATE_BATCH_BUDGET=65536

CSV=./results/beam_all_modes.csv

mkdir -p "$(dirname "$CSV")"
rm -f "$CSV"

export OMP_NUM_THREADS=$THREADS
export OMP_PROC_BIND=close
export OMP_PLACES=cores

for config in \
  "batch-ea|cpu" \
  "batch-ea-blocked|cpu" \
  "batch-ea-blocked|cuda"
do
  IFS="|" read -r mode backend <<< "$config"

  for item in \
    "facebook|$DATA/facebook/facebook.undir.single.txt" \
    "caida|$DATA/caida/caida.undir.single.txt" \
    "email_enron|$DATA/email_enron/email_enron.undir.single.txt" \
    "amazon|$DATA/amazon/amazon.undir.single.txt" \
    "dblp|$DATA/dblp/dblp.undir.single.txt"
  do
    IFS="|" read -r name input <<< "$item"

    echo "===== start name=${name} mode=${mode} backend=${backend} ====="

    "$BEAM" \
      --input "$input" \
      --merge-mode "$mode" \
      --scoring-backend "$backend" \
      --top-k "$TOPK" \
      --ea-use-threshold \
      --threshold-policy reciprocal \
      --iterations "$ITERS" \
      --print-offset 1 \
      --seed "$SEED" \
      --error-bound "$ERROR_BOUND" \
      --group-batch-size "$GROUP_BATCH_SIZE" \
      --candidate-batch-budget "$CANDIDATE_BATCH_BUDGET" \
      --results-csv "$CSV"

    status=$?
    echo "===== end name=${name} mode=${mode} backend=${backend} status=${status} ====="

    if [ "$status" -ne 0 ]; then
      exit "$status"
    fi
  done
done
```
